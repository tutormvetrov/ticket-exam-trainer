- [ ] **Step 4: Прогнать рецензента на трёх моделях**

```python
# scripts/validate_review_engine.py
import json, time, sys
from pathlib import Path
sys.path.insert(0, '.')
from infrastructure.ollama.service import OllamaService
from infrastructure.ollama.client import OllamaClient

svc = OllamaService(OllamaClient(base_url='http://localhost:11434'))

fixtures = Path('tests/fixtures/review_validation')
models = ['qwen3:1.7b', 'qwen3:4b', 'qwen3:8b']

results = {}
for model in models:
    print(f'\n=== Testing {model} ===')
    results[model] = []
    for i in range(1, 6):
        ticket = json.loads((fixtures / f'ticket_{i}.json').read_text(encoding='utf-8'))
        answer = (fixtures / f'answer_{i}.txt').read_text(encoding='utf-8')
        # Формируем reference_theses — для R0 используем просто title как один тезис
        theses = [{'label': 'Основной тезис', 'text': ticket['canonical_answer_summary'][:500]}]
        start = time.time()
        try:
            result = svc.review_answer(ticket['title'], theses, answer, model=model)
            duration = time.time() - start
            valid_json = False
            try:
                parsed = json.loads(result.text)
                valid_json = True
            except Exception:
                parsed = None
            results[model].append({
                'ticket': i, 'duration': duration, 'valid_json': valid_json,
                'raw': result.text[:500]
            })
            print(f'  ticket {i}: {duration:.1f}s, json_ok={valid_json}')
        except Exception as e:
            results[model].append({'ticket': i, 'error': str(e)})
            print(f'  ticket {i}: ERROR {e}')

Path('docs/superpowers/specs/2026-04-18-model-selection-raw.json').write_text(
    json.dumps(results, ensure_ascii=False, indent=2), encoding='utf-8'
)
print('\nRaw results saved.')
```

Run: `python scripts/validate_review_engine.py`

Expected: Скрипт отрабатывает без падения, сохраняет `docs/superpowers/specs/2026-04-18-model-selection-raw.json`. Время на билет для 1.7b: < 90 сек CPU.
