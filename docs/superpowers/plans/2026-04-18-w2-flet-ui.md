```python
"""Font registration for Flet.

Tries Google Fonts URL first (works when online). Falls back to local
TTF files in ui_flet/theme/fonts/ if registered. Classmates behind
corporate firewall get the local fonts.
"""

from pathlib import Path
from typing import Dict

import flet as ft


GOOGLE_FONTS = {
    "Lora": "https://fonts.googleapis.com/css2?family=Lora:wght@400;600&display=swap",
    "Golos Text": "https://fonts.googleapis.com/css2?family=Golos+Text:wght@400;600&display=swap",
    "JetBrains Mono": "https://fonts.googleapis.com/css2?family=JetBrains+Mono:wght@400;600&display=swap",
}

LOCAL_FONTS_DIR = Path(__file__).parent / "fonts"


def register_fonts(page: ft.Page) -> None:
    """Register all fonts on the page. Local TTFs take priority if present."""
    fonts: Dict[str, str] = {}

    for family, url in GOOGLE_FONTS.items():
        local = _local_path_for(family)
        if local and local.exists():
            fonts[family] = str(local.resolve())
        else:
            fonts[family] = url

    page.fonts = fonts


def _local_path_for(family: str) -> Path | None:
    """Map family name to local TTF if available."""
    slug_map = {
        "Lora": "Lora-VariableFont_wght.ttf",
        "Golos Text": "GolosText-VariableFont_wght.ttf",
        "JetBrains Mono": "JetBrainsMono-Regular.ttf",
    }
    slug = slug_map.get(family)
    if not slug:
        return None
    return LOCAL_FONTS_DIR / slug
```

Создать `ui_flet/state.py`:

```python
"""Application state holder for Flet UI.

Holds facade instance, current breakpoint, theme mode, and observer
callbacks. Views subscribe to state changes.
"""

from __future__ import annotations

from dataclasses import dataclass, field
from enum import Enum
from typing import Callable, List, Optional, TYPE_CHECKING

if TYPE_CHECKING:
    from application.facade import AppFacade


class Breakpoint(Enum):
    COMPACT = "compact"
    STANDARD = "standard"
    WIDE = "wide"
    ULTRAWIDE = "ultrawide"
    HUGE_4K = "huge_4k"


@dataclass
class AppState:
    facade: Optional["AppFacade"]
    breakpoint: Breakpoint = Breakpoint.STANDARD
    theme_mode: str = "light"  # "light" | "dark"
    current_route: str = "/library"
    _observers: List[Callable[["AppState"], None]] = field(default_factory=list, repr=False)

    @staticmethod
    def breakpoint_for_width(width: int) -> Breakpoint:
        if width < 1280:
            return Breakpoint.COMPACT
        if width < 1920:
            return Breakpoint.STANDARD
        if width < 2560:
            return Breakpoint.WIDE
        if width < 3840:
            return Breakpoint.ULTRAWIDE
        return Breakpoint.HUGE_4K

    def update_breakpoint(self, width: int) -> None:
        new_bp = self.breakpoint_for_width(width)
        if new_bp != self.breakpoint:
            self.breakpoint = new_bp
            self._notify()

    def toggle_theme(self) -> None:
        self.theme_mode = "dark" if self.theme_mode == "light" else "light"
        self._notify()

    def subscribe(self, observer: Callable[["AppState"], None]) -> None:
        self._observers.append(observer)

    def _notify(self) -> None:
        for obs in self._observers:
            obs(self)
```

```python
"""Route dispatch for Flet views.

Page route changes trigger view swap via registered handlers. Views are
lazy-constructed: only requested route is instantiated on first visit.
"""

from __future__ import annotations

from typing import Callable, Dict, Optional

import flet as ft

from ui_flet.state import AppState


ViewBuilder = Callable[[ft.Page, AppState], ft.Control]


class Router:
    def __init__(self, page: ft.Page, state: AppState):
        self.page = page
        self.state = state
        self._builders: Dict[str, ViewBuilder] = {}
        self._cache: Dict[str, ft.Control] = {}

    def register(self, route: str, builder: ViewBuilder) -> None:
        self._builders[route] = builder

    def navigate(self, route: str) -> Optional[ft.Control]:
        if route not in self._cache:
            builder = self._builders.get(route)
            if builder is None:
                return None
            self._cache[route] = builder(self.page, self.state)
        self.state.current_route = route
        return self._cache[route]
```

```python
"""Base Card component — rounded container with padding and subtle shadow.

Variants:
  - "default" — bg_surface, soft border
  - "elevated" — bg_elevated with shadow
  - "accent" — accent_soft background (for selected/highlighted)
"""

import flet as ft

from ui_flet.theme.tokens import COLOR_LIGHT, RADIUS, SPACE


def Card(
    content: ft.Control,
    variant: str = "default",
    padding: int | None = None,
    on_click: callable | None = None,
) -> ft.Container:
    """Build a Card container.

    Args:
        content: Inner Flet control
        variant: "default" | "elevated" | "accent"
        padding: Override padding (default: SPACE["lg"] = 16)
        on_click: Click handler for interactive cards
    """
    pad = padding if padding is not None else SPACE["lg"]
    palette = COLOR_LIGHT  # при dark theme применяется через page.theme

    bg_by_variant = {
        "default":  palette["bg_surface"],
        "elevated": palette["bg_elevated"],
        "accent":   palette["accent_soft"],
    }

    border_by_variant = {
        "default":  ft.border.all(1, palette["border_soft"]),
        "elevated": None,
        "accent":   ft.border.all(1, palette["accent"]),
    }

    shadow_by_variant = {
        "default":  None,
        "elevated": ft.BoxShadow(blur_radius=8, spread_radius=0, color="#00000015", offset=ft.Offset(0, 2)),
        "accent":   None,
    }

    return ft.Container(
        content=content,
        padding=pad,
        bgcolor=bg_by_variant.get(variant, palette["bg_surface"]),
        border=border_by_variant.get(variant),
        border_radius=RADIUS["lg"],
```

```python
"""Sidebar navigation with 6 routes.

Compact breakpoint: render as drawer (ft.NavigationDrawer). Other: fixed
240-wide column.

from __future__ import annotations

from typing import Callable, List, Tuple

import flet as ft

from ui_flet.i18n import ru as L
from ui_flet.state import AppState, Breakpoint
from ui_flet.theme.tokens import COLOR_LIGHT, RADIUS, SPACE


ROUTES: List[Tuple[str, str, ft.IconValue]] = [
    ("/library",    L.NAV["library"],    ft.Icons.LIBRARY_BOOKS),
    ("/import",     L.NAV["import"],     ft.Icons.UPLOAD_FILE),
    ("/tickets",    L.NAV["tickets"],    ft.Icons.LIST_ALT),
    ("/training",   L.NAV["training"],   ft.Icons.SCHOOL),
    ("/statistics", L.NAV["statistics"], ft.Icons.BAR_CHART),
    ("/settings",   L.NAV["settings"],   ft.Icons.SETTINGS),
]


def Sidebar(
    state: AppState,
    on_navigate: Callable[[str], None],
) -> ft.Control:
    palette = COLOR_LIGHT  # page.theme переопределит в dark

    items = []
    for route, label, icon in ROUTES:
        is_active = state.current_route == route
        items.append(
            ft.Container(
                content=ft.Row([
                    ft.Icon(
                        icon,
                        size=20,
                        color=palette["accent"] if is_active else palette["text_secondary"],
                    ),
                    ft.Text(
                        label,
                        size=14,
                        weight=ft.FontWeight.W_600 if is_active else ft.FontWeight.W_400,
                        color=palette["text_primary"] if is_active else palette["text_secondary"],
                    ),
                ], spacing=SPACE["sm"]),
                padding=ft.padding.symmetric(horizontal=SPACE["lg"], vertical=SPACE["md"]),
                bgcolor=palette["accent_soft"] if is_active else None,
                border=ft.border.only(left=ft.BorderSide(3, palette["accent"])) if is_active else None,
                border_radius=RADIUS["md"],
                on_click=lambda e, r=route: on_navigate(r),
                ink=True,
            )
        )

    return ft.Container(
        content=ft.Column([
            ft.Container(
                content=ft.Text(L.APP_TITLE.split("—")[0].strip(), size=20,
                                weight=ft.FontWeight.BOLD, color=palette["accent"]),
                padding=SPACE["lg"],
            ),
            *items,
        ], spacing=SPACE["xs"]),
        width=240,
        bgcolor=palette["bg_sidebar"],
        border=ft.border.only(right=ft.BorderSide(1, palette["border_soft"])),
    )
```

```python
"""Tezis v2.0 — Flet entry point."""

import flet as ft

from application.facade import AppFacade
from infrastructure.db.connection import connect_to_default_database
from infrastructure.db.repository import Repository
from infrastructure.ollama.client import OllamaClient
from infrastructure.ollama.service import OllamaService

from ui_flet.components.sidebar import Sidebar
from ui_flet.router import Router
from ui_flet.state import AppState, Breakpoint
from ui_flet.theme import build_theme, register_fonts, tokens
from ui_flet.views.library_view import LibraryView
from ui_flet.views.tickets_view import TicketsView
from ui_flet.views.training_view import TrainingView
from ui_flet.views.statistics_view import StatisticsView
from ui_flet.views.import_view import ImportView
from ui_flet.views.settings_view import SettingsView
from ui_flet.i18n import ru as L


def _build_facade() -> AppFacade:
    conn = connect_to_default_database()
    repo = Repository(conn)
    ollama_client = OllamaClient(base_url="http://localhost:11434")
    ollama_service = OllamaService(ollama_client)
    return AppFacade(repository=repo, ollama_service=ollama_service)


def main(page: ft.Page):
    page.title = L.APP_TITLE
    page.window_width = 1280
    page.window_height = 800
    page.window_min_width = 1024
    page.window_min_height = 640

    register_fonts(page)
    page.theme = build_theme(ft.ThemeMode.LIGHT)
    page.dark_theme = build_theme(ft.ThemeMode.DARK)
    page.theme_mode = ft.ThemeMode.LIGHT

    facade = _build_facade()
    state = AppState(facade=facade)

    router = Router(page, state)
    router.register("/library",    lambda p, s: LibraryView(p, s))
    router.register("/tickets",    lambda p, s: TicketsView(p, s))
    router.register("/training",   lambda p, s: TrainingView(p, s))
    router.register("/statistics", lambda p, s: StatisticsView(p, s))
    router.register("/import",     lambda p, s: ImportView(p, s))
    router.register("/settings",   lambda p, s: SettingsView(p, s))

    content_holder = ft.Container(expand=True)

    def navigate(route: str):
        view = router.navigate(route)
        if view is not None:
            content_holder.content = view
        if state.theme_mode == "dark":
            page.theme_mode = ft.ThemeMode.DARK
        else:
            page.theme_mode = ft.ThemeMode.LIGHT
        page.update()

    # Первичный рендер
    navigate("/library")

    # Resize handler → breakpoint
    def on_resize(e):
        state.update_breakpoint(page.window_width)
        # Rebuild current view на breakpoint change
        cached = router._cache.pop(state.current_route, None)
        navigate(state.current_route)

    page.on_resize = on_resize

    page.add(
        ft.Row([
            Sidebar(state, on_navigate=navigate),
            content_holder,
        ], expand=True, spacing=0),
    )


if __name__ == "__main__":
    ft.app(target=main)
```
